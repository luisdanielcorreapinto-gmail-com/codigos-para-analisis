        private static String CadenaConexionADO(String pServidor, String pBD, String pUser = "SA", String pPassword = "",
        Boolean pTrustedConnection = false)
        {

            String mCadenaConexion = String.Empty;

            if (pTrustedConnection)
                mCadenaConexion = "Provider=SQLOLEDB.1;Initial Catalog=" + pBD + ";Data Source=" + pServidor + ";" +
                "Integrated Security=SSPI;";
            else
                mCadenaConexion = "Provider=SQLOLEDB.1;Initial Catalog=" + pBD + ";Data Source=" + pServidor + ";" +
                (pUser.isUndefined() || pPassword.isUndefined() ? "Persist Security Info=False;User ID=" + pUser + ";"
                : "Persist Security Info=True;User ID=" + pUser + ";Password=" + pPassword + ";");

            return mCadenaConexion;

        }

        private static ADODB.Connection ConexionVista(Boolean pADO = true)
        {

            ADODB.Connection mCnLocal = null;

        Retry:

            try
            {

                mCnLocal = new ADODB.Connection();

                mCnLocal.ConnectionTimeout = Properties.Settings.Default.Stellar_ConnectionTimeout;
                //mCnLocal.Provider = "SQLOLEDB.1";
                mCnLocal.ConnectionString = CadenaConexionADO(Properties.Settings.Default.Stellar_SQLServerName,
                "VISTA", Properties.Settings.Default.Stellar_SQLUser,
                Properties.Settings.Default.Stellar_SQLPass, Properties.Settings.Default.Stellar_TrustedConnection);

                mCnLocal.Open();

            }
            catch (System.Runtime.InteropServices.COMException SQLAny)
            {

                //if (SQLAny.Number == 18456 && SQLAny.ErrorCode == -2146232060)
                if (SQLAny.HResult == (-2147217843))
                {

                    dynamic mClsTmp = null;

                    mClsTmp = Functions.SafeCreateObject("SQLSafeGuard.Service");

                    if (mClsTmp == null) goto Otros;

                    System.Windows.Forms.MessageBox.Show("Los datos de acceso para la conexión al Servidor de Stellar " +
                    "no estan establecidos o son incorrectos. Se le solicitarán a continuación.");

                    dynamic TmpVar = mClsTmp.RequestAccess(Program.gCodProducto, Program.gNombreProducto, Program.gPK);

                    if (TmpVar.Length > 0)
                    {
                        Program.mStellarDBUser = TmpVar[0];
                        Program.mStellarDBPass = TmpVar[1];
                        Properties.Settings.Default.Stellar_SQLUser = TmpVar[2];
                        Properties.Settings.Default.Stellar_SQLPass = TmpVar[3];
                        Properties.Settings.Default.Save();
                        goto Retry;
                    }
                    else
                    {
                        Program.Logger.EscribirLog("Los datos de acceso para la conexión al servidor Stellar " +
                        "son incorrectos, Conectando a " + Properties.Settings.Default.Stellar_SQLServerName);
                    }

                }

            Otros:

                Program.Logger.EscribirLog(SQLAny, "Estableciendo conexión al Servidor Stellar.");
                Console.WriteLine(SQLAny);

            }
            catch (Exception Any)
            {
                Program.Logger.EscribirLog(Any, "Estableciendo conexión al Servidor Stellar.");
                Console.WriteLine(Any);
            }

            return mCnLocal;

        }

        private static dynamic ConexionSAP()
        {

            dynamic mCnSAP = null;

            Int32 ErrorNumber; String ErrorDesc;

        Retry:

            try
            {

                mCnSAP = new Company();
                //mCnSAP = Functions.SafeCreateObject("SAPbobsCOM.Company" + Program.DI_API_Version);

                mCnSAP.Server = Properties.Settings.Default.SAP_Server;
                mCnSAP.LicenseServer = Properties.Settings.Default.SAP_LicenseServer;

                mCnSAP.DbServerType = BoDataServerTypes.dst_MSSQL2016;  //BoDataServerTypes.dst_MSSQL2014;
                mCnSAP.DbUserName = Properties.Settings.Default.SAP_DBUser;
                mCnSAP.DbPassword = Properties.Settings.Default.SAP_DBPass;
                mCnSAP.language = BoSuppLangs.ln_Spanish_La; //BoSuppLangs.ln_Spanish_La;
                mCnSAP.UseTrusted = false;

                mCnSAP.CompanyDB = Properties.Settings.Default.SAP_DBName;
                mCnSAP.UserName = Properties.Settings.Default.SAP_CompanyUser;
                mCnSAP.Password = Properties.Settings.Default.SAP_CompanyPass;

                if (mCnSAP.Connect() != 0)
                {

                    mCnSAP.GetLastError(out ErrorNumber, out ErrorDesc);
                    mCnSAP.Disconnect();

                    throw new Exception(ErrorDesc) { HelpLink = ErrorNumber.ToString() };

                }

            }
            catch (Exception SQLAny)
            {

                if (SQLAny.HelpLink == "-4008" || SQLAny.HelpLink == "-132")
                {

                    dynamic mClsTmp = null;

                    mClsTmp = Functions.SafeCreateObject("SQLSafeGuard.Service");

                        "son incorrectos, Conectando a [" + Properties.Settings.Default.SAP_Server + "]");
                        InsertarAuditoria(Program.LogIDAuditoria, Program.ErrorLog, "Los datos de acceso para la conexión de base de datos al Servidor SAP " +
                        "son incorrectos, Conectando a [" + Properties.Settings.Default.SAP_Server + "]", "DBSync.ConexionSAP()",
                        "Servidor SAP", Properties.Settings.Default.SAP_Server,
                        (SQLAny.HelpLink + "|" + SQLAny.Message), "ConexionSAP", Program.mCnLocal);
                    }

                    /*}
                    else if (SQLAny.HelpLink == "-8023")
                    {*/

                    //dynamic mClsTmp = null;
                    mClsTmp = null;

                    mClsTmp = Functions.SafeCreateObject("SQLSafeGuard.Service");

                    if (mClsTmp == null) goto Otros;

                    System.Windows.Forms.MessageBox.Show("Los datos de acceso para la conexión a la Compañia SAP " +
                    "no estan establecidos o son incorrectos. Se le solicitarán a continuación.");

                    //dynamic 
                    TmpVar = mClsTmp.RequestAccess(Program.gCodProducto, Program.gNombreProducto, Program.gPK);

                    {
                        Program.mSAPUserLogin = TmpVar[0];

                        Properties.Settings.Default.SAP_CompanyPass = TmpVar[3];

                        "Servidor SAP", Properties.Settings.Default.SAP_Server,

            Otros:

                Program.Logger.EscribirLog(SQLAny, "Estableciendo conexión al Servidor SAP.");
                InsertarAuditoria(Program.LogIDAuditoria, Program.ErrorLog,
                "Estableciendo conexión al Servidor SAP.", "DBSync.ConexionSAP()", "Servidor SAP",

                    Program.Logger.EscribirLog("No se pudo establecer conexión al Servidor SAP.");
                    System.Environment.Exit(0);
                }
            }
            else
            {
                Program.Logger.EscribirLog("Verifique que SAP BUSINESS ONE y la DI API esten correctamente instalados.");
                System.Environment.Exit(0);
            }

            return mCnSAP;

        }

                                "DBSync.IniciarAgente()", "Ejecucion Agente", Program.gCorrelativo, (ex.HResult + "|" + ex.Message), "Validacion Stellar", Program.mCnLocal);
                                return;
                            }

                        }

                        //nRegistrosLote = Properties.Settings.Default.nRegistrosLote;
                        //nLotesMaxEjecucion = Properties.Settings.Default.nLotesCorrida;

                        //Boolean result = PrepararBD(ref mCnLocal);

                        //SincronizarVentas();

                    }
                    else
                    {
                        throw new Exception("Falla al obtener correlativo de ejecución de Agente.");
                    }

                }

            }
            catch (System.Exception Any)
            {
                Program.Logger.EscribirLog(Any, "Ejecutando Agente.");
                InsertarAuditoria(Program.LogIDAuditoria, Program.ErrorLog,
                "Ejecutando Agente.", "DBSync.IniciarAgente()", "Corrida Agente",
                Program.gCorrelativo, Program.mCnLocal);
            }
            finally
            {

                if (SAPCon != null)
                    if (SAPCon.Connected)
                        SAPCon.Disconnect();

                if (mCnLocal != null)
                    if (mCnLocal.State != Convert.ToInt32(ADODB.ObjectStateEnum.adStateClosed))
                        mCnLocal.Close();

            }

        }

                            LocalRs.Fields["Cant_Decimales"].Value = mRs.Fields.Item("Cant_Decimales").Value;

                        if (Convert.ToInt32(LocalRs.Fields["n_TipoPeso"].Value) == 4)
                            LocalRs.Fields["nu_NivelClave"].Value = 1;

                        LocalRs.Fields["c_Observacio"].Value = mDatoTemp.Replace("\r\r", System.Environment.NewLine);
                        LocalRs.Fields["c_Seriales"].Value = 1; // No maneja seriales a lo interno pero en el POS para la funcionalidad los va a pedir.
                        LocalRs.Fields["n_TipoPeso"].Value = 0; // Al ser un producto que pida seriales solo puede ser unitario.

        private static string ObtenerWhsCodeTienda(string Localidad)
        {
            try
            {
                var TiendaStellar = StoresList.FirstOrDefault(x => x.Name == Localidad);

                return TiendaStellar.WhsCode;
            }
            catch (Exception ex)
            {
                return "";
            }


        }
        private static string ObtenerComplejoTienda(string Localidad)
        {
            try
            {
                var TiendaStellar = StoresList.FirstOrDefault(x => x.Name == Localidad);

                return TiendaStellar.Filler;
            }
            catch (Exception ex)
            {
                return "";
            }


        }

        private static void ActualizarTipodeCambio()
        {

            SAPbobsCOM.Company SAPDB = Program.mCnSAP;
            ADODB.Connection LocalDB = Program.mCnLocal;



            String mEtapa = String.Empty;

            Object Records; Boolean ActiveTrans = false;

            try
            {

                mEtapa = "Limpiando tablas temporales";
                Console.WriteLine(mEtapa);
                if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);

                try
                {
                    LocalDB.Execute("DROP TABLE #TMP_MONEDAS_SAP", out Records);
                }
                catch { }

                double TasaAnterior = 0;

                mEtapa = "Creando tablas temporales";
                Console.WriteLine(mEtapa);
                if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);

                LocalDB.Execute("SELECT c_codmoneda,  c_descripcion" +
                    " ,n_factor,b_preferencia ,c_observacio ,b_activa ,c_simbolo ,n_decimales ,bUsoEnPOS ,CodigoISO ,n_Porc_IGTF ,c_Modalidad_IGTF INTO #TMP_MONEDAS_SAP FROM MA_MONEDAS WHERE (c_codmoneda='USD' OR c_codmoneda='EUR' ) ", out Records);

                mEtapa = "Creando Queries SAP";
                Console.WriteLine(mEtapa);
                if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);

                String SQLMonedas = String.Empty +
                "SELECT [RateDate] , [Currency] AS [c_codmoneda], [Rate] AS [n_factor] FROM ORTT WHERE ([Currency]='USD' OR [Currency]='EUR') AND [RateDate] =  '" + DateTime.Now.ToString("yyyy-dd-MM") + "' ORDER BY [Currency]";

                SAPbobsCOM.Recordset mRs;
                String mDatoTemp;

                //

                mEtapa = "Consultando Monedas en SAP";
                Console.WriteLine(mEtapa);
                if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);
                List<Currency> CurrencyList = new List<Currency>();
                mRs = SAPDB.GetBusinessObject(BoObjectTypes.BoRecordset);

                mRs.DoQuery(SQLMonedas);

                if (!mRs.EoF)
                {

                    mEtapa = "Insertando Registros Temp Monedas";
                    Console.WriteLine(mEtapa);
                    if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);

                    ADODB.Recordset LocalRs = new ADODB.Recordset();

                    LocalRs.CursorLocation = ADODB.CursorLocationEnum.adUseClient;

                    LocalRs.Open("SELECT * FROM #TMP_MONEDAS_SAP ORDER BY c_codmoneda", LocalDB, ADODB.CursorTypeEnum.adOpenDynamic, ADODB.LockTypeEnum.adLockBatchOptimistic);

                    LocalRs.ActiveConnection = null;


                    while (!mRs.EoF)
                    {

                        TasaAnterior = Math.Round(Convert.ToDouble(LocalRs.Fields["n_factor"].Value), 2);

                        LocalRs.AddNew();

                        Currency Currency = new Currency();

                        mDatoTemp = mRs.Fields.Item("c_codmoneda").Value.ToString();
                        LocalRs.Fields["c_codmoneda"].Value = mDatoTemp.Left(LocalRs.Fields["c_codmoneda"].DefinedSize);
                        LocalRs.Fields["n_factor"].Value = Math.Round(mRs.Fields.Item("n_factor").Value, 2);

                        LocalRs.Fields["b_preferencia"].Value = 0;
                        LocalRs.Fields["c_observacio"].Value = "";
                        LocalRs.Fields["b_activa"].Value = 1;
                        LocalRs.Fields["c_simbolo"].Value = "";
                        LocalRs.Fields["n_decimales"].Value = 2;
                        LocalRs.Fields["bUsoEnPOS"].Value = 1;
                        LocalRs.Fields["CodigoISO"].Value = "";
                        LocalRs.Fields["n_Porc_IGTF"].Value = 3;
                        LocalRs.Fields["c_Modalidad_IGTF"].Value = "3";

                        Currency.c_codmoneda = LocalRs.Fields["c_codmoneda"].Value.ToString();
                        Currency.n_factor = Math.Round(Convert.ToDouble(LocalRs.Fields["n_factor"].Value), 2);
                        Currency.b_preferencia = Convert.ToInt32(LocalRs.Fields["b_preferencia"].Value);
                        Currency.c_observacio = LocalRs.Fields["c_observacio"].Value.ToString();
                        Currency.b_activa = Convert.ToInt32(LocalRs.Fields["b_preferencia"].Value);
                        Currency.c_simbolo = LocalRs.Fields["c_simbolo"].Value.ToString();
                        Currency.n_decimales = Convert.ToInt32(LocalRs.Fields["n_decimales"].Value);
                        Currency.bUsoEnPOS = Convert.ToInt32(LocalRs.Fields["bUsoEnPOS"].Value);
                        Currency.CodigoISO = LocalRs.Fields["CodigoISO"].Value.ToString();
                        Currency.n_Porc_IGTF = Convert.ToDouble(LocalRs.Fields["n_Porc_IGTF"].Value);
                        Currency.c_Modalidad_IGTF = LocalRs.Fields["c_Modalidad_IGTF"].Value.ToString();

                        CurrencyList.Add(Currency);

                        mRs.MoveNext();

                    }



                    // LocalRs.ActiveConnection = LocalDB;

                    //LocalRs.UpdateBatch();

                }


                mEtapa = "Insertando registros TR_PEND a partir de tablas Tmp";
                Console.WriteLine(mEtapa);
                if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);


                mEtapa = "Consultando Tasa Anterior Historicos de Monedas";
                Console.WriteLine(mEtapa);
                if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);


                foreach (var Currency in CurrencyList)
                {
                    try
                    {
                        ADODB.Recordset LocalRS = new ADODB.Recordset();

                        LocalRS.CursorLocation = ADODB.CursorLocationEnum.adUseClient;

                        LocalRS.Open("SELECT MAX(d_FechaCambioAnterior) AS 'd_FechaCambioAnterior' FROM MA_HISTORICO_MONEDAS WHERE c_CodMoneda=" + "'" + Currency.c_codmoneda + "'", LocalDB, ADODB.CursorTypeEnum.adOpenDynamic, ADODB.LockTypeEnum.adLockBatchOptimistic);

                        LocalRS.ActiveConnection = null;

                        DateTime d_FechaCambioAnterior = DateTime.Now;

                        while (!LocalRS.EOF)
                        {
                            var FechaCambio = LocalRS.Fields["d_FechaCambioAnterior"].Value.ToString();

                            if (FechaCambio != "")
                            {
                                d_FechaCambioAnterior = Convert.ToDateTime(LocalRS.Fields["d_FechaCambioAnterior"].Value.ToString());

                            }


                            LocalRS.MoveNext();
                        }


                        LocalDB.BeginTrans(); ActiveTrans = true;

                        mEtapa = "Insertando Actualizaciones de Monedas";
                        Console.WriteLine(mEtapa);
                        if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);


                        LocalDB.Execute(
                 "UPDATE [MA_MONEDAS] \n" +
                 "SET [n_factor]= " + Currency.n_factor + " WHERE [c_codmoneda]= '" + Currency.c_codmoneda + "'" + " \n"

                 , out Records);

                        InsertarAuditoria(10002, Program.InfoLog,
                            "Moneda " + Currency.c_codmoneda + " actualizada con exito. Factor " + Currency.n_factor, "DBSync.ActualizarMonedas()",
                            "POSTransaccionEnviada", Currency.c_codmoneda, "0", "EnviarTransaccionPOS", Program.mCnLocal);


                        LocalDB.Execute(
                  "INSERT INTO MA_HISTORICO_MONEDAS (d_FechaCambioAnterior, d_FechaCambioActual, c_CodMoneda, c_Descripcion,n_FactorAnterior,n_FactorPeriodo,b_Preferencia,c_Observacio,b_Activa,c_Simbolo,n_Decimales,FechaInsert,Host_Name,c_CodUsuario)\n" +
                  "SELECT '" + d_FechaCambioAnterior.ToString("yyyy-dd-MM") + "' AS d_FechaCambioAnterior, '" + DateTime.Now.ToString("yyyy-dd-MM") + "' AS d_FechaCambioActual, TMP.c_codmoneda, TMP.c_descripcion, " + TasaAnterior + " AS n_FactorAnterior, TMP.n_factor, TMP.b_preferencia, TMP.c_observacio, TMP.b_activa, TMP.c_simbolo, TMP.n_decimales, GETDATE(), 'SRVSTELLARLAB|Stellar BUSINESS' AS 'Host_Name', '312181230' AS 'c_CodUsuario'  \n" +
                  "FROM MA_MONEDAS TMP WHERE TMP.c_codmoneda='" + Currency.c_codmoneda + "'"
                  , out Records);

                        InsertarAuditoria(10002, Program.InfoLog,
                            "Historico de monedas para " + Currency.c_codmoneda + " actualizada con exito. Factor " + Currency.n_factor + " Fecha: " + DateTime.Now.ToString("yyyy-dd-MM"), "DBSync.ActualizarMonedas()",
                            "POSTransaccionEnviada", Currency.c_codmoneda, "0", "EnviarTransaccionPOS", Program.mCnLocal);



                        LocalDB.Execute(
                        "INSERT INTO TR_PEND_MONEDAS \n" +
                        "(C_CODMONEDA, C_DESCRIPCION, N_FACTOR, B_PREFERENCIA,C_OBSERVACIO, B_ACTIVA, C_SIMBOLO, N_DECIMALES, TIPO_CAMBIO, bUsoEnPOS, CodigoISO, n_Porc_IGTF, c_Modalidad_IGTF) \n" +
                        "SELECT TMP.c_codmoneda, TMP.c_descripcion, TMP.n_factor, TMP.b_preferencia, TMP.c_observacio, TMP.b_activa, TMP.c_simbolo, TMP.n_decimales, 0 AS TIPO_CAMBIO, TMP.bUsoEnPOS, TMP.CodigoISO, TMP.n_Porc_IGTF, TMP.c_Modalidad_IGTF \n" +
                        "FROM MA_MONEDAS TMP WHERE TMP.c_codmoneda='" + Currency.c_codmoneda + "'"
                        , out Records);


                        LocalDB.Execute(
                  "INSERT INTO TR_PEND_HISTORICO_MONEDAS  (d_FechaCambioAnterior, d_FechaCambioActual, c_CodMoneda, c_Descripcion,n_FactorAnterior,n_FactorPeriodo,b_Preferencia,c_Observacio,b_Activa,c_Simbolo,n_Decimales,FechaInsert,Host_Name,c_CodUsuario, TipoCambio)\n" +
                  "SELECT '" + d_FechaCambioAnterior.ToString("yyyy-dd-MM") + "' AS d_FechaCambioAnterior, '" + DateTime.Now.ToString("yyyy-dd-MM") + "' AS d_FechaCambioActual, TMP.c_codmoneda, TMP.c_descripcion, " + TasaAnterior + " AS n_FactorAnterior, TMP.n_factor, TMP.b_preferencia, TMP.c_observacio, TMP.b_activa, TMP.c_simbolo, TMP.n_decimales, GETDATE(), 'SRVSTELLARLAB|Stellar BUSINESS' AS 'Host_Name', '312181230' AS 'c_CodUsuario', 1 AS 'TipoCambio'  \n" +
                  "FROM MA_MONEDAS TMP WHERE TMP.c_codmoneda='" + Currency.c_codmoneda + "'"
                  , out Records);

                        LocalDB.CommitTrans(); ActiveTrans = false;

                        InsertarAuditoria(10002, Program.InfoLog,
                           "TR_PEND MONEDAS insertados correctamente", "", "0", "EnviarTransaccionPOS", Program.mCnLocal);

                    }
                    catch (Exception Any)
                    {

                        if (ActiveTrans) LocalDB.RollbackTrans();

                        Console.WriteLine("Error: " + mEtapa);

                        if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);


                        InsertarAuditoria(10002, Program.ErrorLog,
                        "Error durante el procesamiento de Actualizacion de Monedas." + "\t" +
                        " Fecha " + DateTime.Now + "\t",
                        "ActualizarMonedas()", "Agente SxSBO", "",
                        ("[" + Any.HResult + "][" + Any.Message + "]"), "Procesando_Excepcion", Program.mCnLocal);

                        Program.Logger.EscribirLog(Any, "ActualizarMonedas(); Etapa: " + mEtapa);

                    }

                }

                //LocalDB.CommitTrans(); ActiveTrans = false;

                mEtapa = "Insercion de Pendientes de Monedas finalizado.";
                Console.WriteLine(mEtapa);
                if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);

            }
            catch (Exception Any)
            {

                if (ActiveTrans) LocalDB.RollbackTrans();

                Console.WriteLine("Error: " + mEtapa);

                if (Properties.Settings.Default.DebugMode) MessageBox.Show(mEtapa);

                InsertarAuditoria(10002, Program.ErrorLog,
                        "Error durante el procesamiento de Actualizacion de Monedas." + "\t" +
                        " Fecha " + DateTime.Now + "\t",
                        "ActualizarMonedas()", "Agente SxSBO", "",
                        ("[" + Any.HResult + "][" + Any.Message + "]"), "Procesando_Excepcion", Program.mCnLocal);

                Program.Logger.EscribirLog(Any, "ActualizarMonedas(); Etapa: " + mEtapa);

            }

        }

    }
